Resizing

When resizing, an optional Boolean parameter indicated whether previous entries should be preserved or not.

#include <Array/ArrayIo.h>
#include <Array/ArrayOp.h>

int main(int argc, char** argv)
{
    Array<double> a, b, c;
    a.resize(3u);
    b.resize(3u);

    for(unsigned i=0; i<b.dim(0); i++) b(i)=(i+1)/10.;
    writeArray(b, std::cout);	

    b.resize(5, true); // keep elements
    writeArray(b, std::cout);	

    b.resize(7, false); // delete existing entries
    writeArray(b, std::cout);	

    return EXIT_SUCCESS;
}

The output should looks like this:

0.1 0.2 0.3 0.1 0.2 0.3 0 0 0 0 0 0 0 0 0
