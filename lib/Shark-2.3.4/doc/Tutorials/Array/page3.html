Basic Numerical Operations

Having included Array/ArrayOp.h, we can now apply basic numerical operations, for example adding two vectors. 

#include <Array/ArrayIo.h>
#include <Array/ArrayOp.h>

int main(int argc, char** argv)
{
    Array<double> a, b, c;
    a.resize(3u);
    b.resize(3u);

    a = 1.;
    for(unsigned i=0; i<b.dim(0); i++) b(i)=(i+1)/10.;

    c = a + b;
    writeArray(c, std::cout);	

    return EXIT_SUCCESS;
}
