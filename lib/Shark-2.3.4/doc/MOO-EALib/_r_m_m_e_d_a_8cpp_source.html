01107 //Householder reduction of Matrix a to tridiagonal form.
01108 //
01109 // Algorithm: Martin et al., Num. Math. 11, 181-195, 1968.
01110 // Ref: Smith et al., Matrix Eigensystem Routines -- EISPACK Guide
01111 // Springer-Verlag, 1976, pp. 489-494.
01112 // W H Press et al., Numerical Recipes in C, Cambridge U P,
01113 // 1988, pp. 373-374. 
01114 void Matrix::tred2(FVECTOR& eigenvalue, FVECTOR& interm, Matrix& eigenvector)
01115 {
01116     double scale, hh, h, g, f;
01117     int k, j, i,l;
01118 
01119     for(i = mRow-1; i >= 1; i--)
01120     {
01121         l = i - 1;
01122         h = scale = 0.0;
01123         if(l > 0)
01124         {
01125             for (k = 0; k <= l; k++)
01126                 scale += fabs(eigenvector(i, k));
01127             if (scale == 0.0)
01128                 interm[i] = eigenvector(i, l);
01129             else
01130             {
01131                 for (k = 0; k <= l; k++)
01132                 {
01133                     eigenvector(i, k) /= scale;
01134                     h += eigenvector(i, k) * eigenvector(i, k);
01135                 }
01136                 f = eigenvector(i, l);
01137                 g = f>0 ? -sqrt(h) : sqrt(h);
01138                 interm[i] = scale * g;
01139                 h -= f * g;
01140                 eigenvector(i, l) = f - g;
01141                 f = 0.0;
01142                 for (j = 0; j <= l; j++)
01143                 {
01144                     eigenvector(j, i) = eigenvector(i, j)/h;
01145                     g = 0.0;
01146                     for (k = 0; k <= j; k++)
01147                         g += eigenvector(j, k) * eigenvector(i, k);
01148                     for (k = j+1; k <= l; k++)
01149                         g += eigenvector(k, j) * eigenvector(i, k);
01150                     interm[j] = g / h;
01151                     f += interm[j] * eigenvector(i, j);
01152                 }
01153                 hh = f / (h + h);
01154                 for (j = 0; j <= l; j++)
01155                 {
01156                     f = eigenvector(i, j);
01157                     interm[j] = g = interm[j] - hh * f;
01158                     for (k = 0; k <= j; k++)
01159                         eigenvector(j, k) -= (f * interm[k] + g * eigenvector(i, k));
01160                 }
01161             }
01162         }
01163         else
01164             interm[i] = eigenvector(i, l);
01165         eigenvalue[i] = h;
01166     }
01167     eigenvalue[0] = 0.0 ;   
01168     interm[0] = 0.0;
01169     for(i = 0; i < (int)mRow; i++)
01170     {
01171         l = i - 1;
01172         if(eigenvalue[i])
01173         {
01174             for (j = 0; j <= l; j++)
01175             {
01176                 g = 0.0;
01177                 for (k = 0; k <= l; k++)
01178                     g += eigenvector(i, k) * eigenvector(k, j);
01179                 for (k = 0; k <= l; k++)
01180                     eigenvector(k, j) -= g * eigenvector(k, i);
01181             }
01182         }
01183         eigenvalue[i] = eigenvector(i, i);
01184         eigenvector(i, i) = 1.0;
01185         for (j = 0; j <= l; j++)
01186             eigenvector(j, i) = eigenvector(i, j) = 0.0;
01187     }

01191 //Tridiagonal QL algorithm -- Implicit 
01192 void Matrix::tqli(FVECTOR& eigenvalue, FVECTOR& interm, Matrix& eigenvector)
01193 {
01194     int m, l, iter, i, k;
01195     double s, r, p, g, f, dd, c, b;
01196     
01197     for (i = 1; i < (int)mRow; i++)
01198         interm[i-1] = interm[i];
01199     interm[mRow-1] = 0.0;
01200     for (l = 0; l < (int)mRow; l++)
01201     {
01202         iter = 0;
01203         do
01204         {
01205             if (iter++ > 100) break;
01206 
01207             for (m = l; m < (int)mRow-1; m++)
01208             {
01209                 dd = fabs(eigenvalue[m]) + fabs(eigenvalue[m+1]);
01210                 if (fabs(interm[m]) + dd == dd || !wxFinite(dd)) break;
01211             }
01212             if (m != l)
01213             {
01214                 //if (iter++ == 30) erhand("No convergence in TLQI.");
01215                 g = (eigenvalue[l+1] - eigenvalue[l]) / (2.0 * interm[l]);
01216                 r = sqrt((g * g) + 1.0);
01217                 g = eigenvalue[m] - eigenvalue[l] + interm[l] / (g + SIGN(r, g));
01218                 s = c = 1.0;
01219                 p = 0.0;
01220                 for (i = m-1; i >= l; i--)
01221                 {
01222                     f = s * interm[i];
01223                     b = c * interm[i];
01224                     if (fabs(f) >= fabs(g))
01225                     {
01226                         c = g / f;
01227                         r = sqrt((c * c) + 1.0);
01228                         interm[i+1] = f * r;
01229                         c *= (s = 1.0/r);
01230                     }
01231                     else
01232                     {
01233                         s = f / g;
01234                         r = sqrt((s * s) + 1.0);
01235                         interm[i+1] = g * r;
01236                         s *= (c = 1.0/r);
01237                     }
01238                     g = eigenvalue[i+1] - p;
01239                     r = (eigenvalue[i] - g) * s + 2.0 * c * b;
01240                     p = s * r;
01241                     eigenvalue[i+1] = g + p;
01242                     g = c * r - b;
01243                     for (k = 0; k < (int)mRow; k++)
01244                     {
01245                         f = eigenvector(k, i+1);
01246                         eigenvector(k, i+1)= s * eigenvector(k, i) + c * f;
01247                         eigenvector(k, i) = c * eigenvector(k, i) - s * f;
01248                     }
01249                 }
01250                 eigenvalue[l] = eigenvalue[l] - p;
01251                 interm[l] = g;
01252                 interm[m] = 0.0;
01253             }
01254         }  while (m  != l);
01255     }
