01003 
01004 double DebRotatedF1(const std::vector< double > &x, Array<double> &coord)
01005 {
01006     double f1;//, penalty = 0;
01007     f1 = scalarprod(coord.col(0), x);
01008 
01009     // Penalty
01010     if (f1 > .3 || f1 < -.3)
01011         //penalty = fabs(f1) - .3;
01012         return MAXDOUBLE;
01013     //
01014     return f1 ;//+ penalty;
01015 }

01018 //
01019 double DebRotatedF2(const std::vector< double >& x, Array<double> &coord)
01020 {
01021     double   sum = 0.0;
01022     unsigned n;
01023     double   g, f1, f2;//, penalty=0;
01024     unsigned i;
01025 
01026     n  = x.size();
01027 
01028     Array<double> v(n);
01029 
01030     for (i = 0; i < n; i++)
01031         v(i) = scalarprod(coord.col(i), x);
01032 
01033     f1 = v(0);
01034 
01035     if (f1 > .3 || f1 < -.3)
01036         //penalty = fabs(f1) - .3;
01037         return MAXDOUBLE;
01038 
01039     //
01040     for (i = 1; i < n; i++)
01041         sum += v(i) * v(i) - 10. * cos(4 * M_PI * v(i));
01042 
01043     g = 1.0 + 10. * (double)(n - 1) + sum;
01044 
01045     f2 = g * exp(- 1. * (f1 / g));
01046 
01047     return f2 ;//+ penalty;
01048 }
