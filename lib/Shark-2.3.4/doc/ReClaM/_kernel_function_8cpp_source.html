00752 double NormalizedKernel::evalDerivative(const Array<double>& x1, const Array<double>& x2, Array<double>& derivative) const
00753 {
00754     int i, ic = getParameterDimension();
00755     Array<double> der11(ic);
00756     Array<double> der12(ic);
00757     Array<double> der22(ic);
00758 
00759     // compute the single kernel values with derivatives
00760     double k11 = baseKernel->evalDerivative(x1, x1, der11);
00761     double k12 = baseKernel->evalDerivative(x1, x2, der12);
00762     double k22 = baseKernel->evalDerivative(x2, x2, der22);
00763     double s = sqrt(k11 * k22);
00764 
00765     // compute the derivative using the quotient rule
00766     derivative.resize(ic, false);
00767     for (i = 0; i < ic; i++)
00768     {
00769         derivative(i) = (der12(i) * k11 * k22 - 0.5 * k12 * (k11 * der22(i) + der11(i) * k22)) / (s * k11 * k22);
00770     }
00771 
00772     // return the normalized kernel value
00773     return k12 / s;
00774 }
