00074 // alpha, the gradient g, and the quadratic term Q in the
00075 // interval [L, U]. Return the corresponding gain.
00076 double QpMcDecomp::StepEdge(double alpha, double g, double Q, double L, double U, double& mu)
00077 {
00078     // compute the optimal unconstrained step
00079     double muHat = g / Q;
00080 
00081     // compute the optimal constrained step
00082     if (muHat < L - alpha) mu = L - alpha;
00083     else if (muHat > U - alpha) mu = U - alpha;
00084     else mu = muHat;
00085 
00086     // compute (twice) the gain
00087     if (! finite(muHat)) return 1e100;
00088     double deltaMu = muHat - mu;
00089     return (muHat * muHat - deltaMu * deltaMu) * Q;
00090 }

00708         // exchange examples such that shrinked examples
00709         // are moved to the ends of the lists
00710         for (a = activeEx - 1; a >= 0; a--)
00711         {
00712             if (example[a].active == 0) DeactivateExample(a);
00713         }
00714 
00715         // shrink the corresponding cache entries
00716         for (a = 0; a < (int)activeEx; a++)
00717         {
00718             if (kernelMatrix.getCacheRowSize(a) > activeEx) kernelMatrix.CacheRowResize(a, activeEx);
00719         }
00720     }
00721 }

00723 void QpMcDecomp::Unshrink(bool complete)
00724 {
00725     if (activeVar == variables) return;
00726 
00727     unsigned int i, a;
00728     float* q;
00729     double v;
00730 
00731     // compute the inactive gradient components (quadratic time complexity)
00732     for (a = activeVar; a < variables; a++) gradient(a) = linear(a);
00733     for (i = 0; i < variables; i++)
00734     {
00735         v = alpha(i);
00736         if (v == 0.0) continue;
00737 
00738         unsigned int ei = variable[i].example;
00739         unsigned int vi = variable[i].label;
00740         unsigned int yi = example[ei].label;
00741         q = kernelMatrix.Row(ei, 0, examples, true);
00742         for (a = activeVar; a < variables; a++)
00743         {
00744             unsigned int ea = variable[a].example;
00745             unsigned int va = variable[a].label;
00746             unsigned int ya = example[ea].label;
00747 
00748             double km = Modifier(yi, ya, vi, va);
00749             double k = q[ea];
00750             gradient(a) -= km * k * v;
00751         }
00752     }
00753 
00754     for (i = 0; i < examples; i++) example[i].active = classes;
00755     activeEx = examples;
00756     activeVar = variables;
00757 
00758     if (! complete) Shrink();
00759 }

00761 void QpMcDecomp::DeactivateVariable(unsigned int v)
00762 {
00763     unsigned int ev = variable[v].example;
00764     unsigned int iv = variable[v].index;
00765     tExample* exv = &example[ev];
00766     unsigned int ih = exv->active - 1;
00767     unsigned int h = exv->variables[ih];
00768     variable[v].index = ih;
00769     variable[h].index = iv;
00770     std::swap(exv->variables[iv], exv->variables[ih]);
00771     iv = ih;
00772     exv->active--;
00773 
00774     unsigned int j = activeVar - 1;
00775     unsigned int ej = variable[j].example;
00776     unsigned int ij = variable[j].index;
00777     tExample* exj = &example[ej];
00778 
00779     // exchange entries in the lists
00780     XCHG_A(double, boxMin, v, j);
00781     XCHG_A(double, boxMax, v, j);
00782     XCHG_A(double, linear, v, j);
00783     XCHG_A(double, alpha, v, j);
00784     XCHG_A(double, gradient, v, j);
00785     XCHG_V(tVariable, variable, v, j);
00786 
00787     std::swap(exv->variables[iv], exj->variables[ij]);
00788 
00789     activeVar--;
00790 }
