01023 const Array<double>& C_SVM::PrepareDerivative()
01024 {
01025     if (norm2penalty) throw SHARKEXCEPTION("[C_SVM::PrepareDerivative] PrepareDerivative works only in the 1-norm case.");
01026 
01027     SVM* svm = getSVM();
01028     const Array<double>& pt = svm->getPoints();
01029     int a, ac = svm->getExamples();
01030     int s, sc = svm->getParameterDimension();
01031     int k, kc = kernel->getParameterDimension();
01032     int pc = getParameterDimension();
01033     Array<double> der;
01034 
01035     alpha_b_Derivative.resize(sc, pc, false);
01036     alpha_b_Derivative = 0.0;
01037 
01038     // determine free and bounded support vectors
01039     std::vector<int> fi;
01040     std::vector<int> ri;
01041     for (a=0; a<ac; a++)
01042     {
01043         double alpha = svm->getAlpha(a);
01044         if (alpha != 0.0)
01045         {
01046             if ((alpha == -C_minus) || (alpha == C_plus)) ri.push_back(a);
01047             else fi.push_back(a);
01048         }
01049     }
01050     int f, fc = fi.size();
01051     int r, rc = ri.size();
01052     fi.push_back(ac);
01053 
01054     Array2D<double> H(fc+1, fc+1);
01055     Array2D<double> H_inv(fc+1, fc+1);
01056     Array2D<double> R(fc+1, rc);
01057     Array<double> dH(kc, fc+1, fc+1);
01058     Array<double> dR(kc, fc+1, rc);
01059     Array<double> alpha_f(fc+1);
01060     Array<double> alpha_r(rc);
01061     for (f=0; f<fc; f++) alpha_f(f) = svm->getAlpha(fi[f]);
01062     alpha_f(fc) = svm->getOffset();
01063     for (r=0; r<rc; r++) alpha_r(r) = svm->getAlpha(ri[r]);
01064 
01065     // initialize H and dH
01066     for (f=0; f<fc; f++)
01067     {
01068         int f2;
01069         for (f2=0; f2<f; f2++)
01070         {
01071             H(f, f2) = H(f2, f) = kernel->evalDerivative(pt[fi[f]], pt[fi[f2]], der);
01072             for (k=0; k<kc; k++)
01073             {
01074                 dH(k, f, f2) = dH(k, f2, f) = der(k);
01075             }
01076         }
01077         H(f, f) = kernel->evalDerivative(pt[fi[f]], pt[fi[f]], der);
01078         for (k=0; k<kc; k++)
01079         {
01080             dH(k, f, f) = der(k);
01081         }
01082         H(fc, f) = H(f, fc) = 1.0;
01083         for (k=0; k<kc; k++) dH(k, fc, f) = dH(k, f, fc) = 0.0;
01084     }
01085     H(fc, fc) = 0.0;
01086     for (k=0; k<kc; k++) dH(k, fc, fc) = 0.0;
01087 
01088     // compute H_inv
01089 //  invertSymm(H_inv, H);
01090     g_inverse(H, H_inv, 200, 1e-10, true);
01091 
01092     // initialize R and dR
01093     for (r=0; r<rc; r++)
01094     {
01095         for (f=0; f<fc; f++)
01096         {
01097             R(f, r) = kernel->evalDerivative(pt[fi[f]], pt[ri[r]], der);
01098             for (k=0; k<kc; k++)
01099             {
01100                 dR(k, f, r) = der(k);
01101             }
01102         }
01103         R(fc, r) = 1.0;
01104         for (k=0; k<kc; k++) dR(k, fc, r) = 0.0;
01105     }
01106 
01107     der.resize(fc+1, false);
01108 
01109     // compute the derivative of (\alpha, b) w.r.t. C
01110     if (rc > 0)
01111     {
01112         Array<double> y(rc);
01113         for (r=0; r<rc; r++) y(r) = ((alpha_r(r) > 0.0) ? 1.0 : -1.0);
01114         Array<double> Ry(fc+1);
01115         matColVec(Ry, R, y);
01116         matColVec(der, H_inv, Ry);
01117         for (f=0; f<=fc; f++) alpha_b_Derivative(fi[f], 0) = -der(f);
01118         for (r=0; r<rc; r++) alpha_b_Derivative(ri[r], 0) = ((alpha_r(r) > 0.0) ? 1.0 : -C_ratio);
01119         if (exponential)
01120         {
01121             for (s=0; s<sc; s++) alpha_b_Derivative(s, 0) *= C_plus;
01122         }
01123     }
01124 
01125     // compute the derivative of (\alpha, b) w.r.t. the kernel parameters
01126     for (k=0; k<kc; k++)
01127     {
01128         Array<double> sum(fc+1);
01129         Array<double> tmp(fc+1);
01130         matColVec(sum, dH[k], alpha_f);
01131         matColVec(tmp, dR[k], alpha_r);
01132         for (f=0; f<=fc; f++) sum(f) += tmp(f);
01133         matColVec(der, H_inv, sum);
01134         for (f=0; f<=fc; f++) alpha_b_Derivative(fi[f], k+1) = -der(f);
01135     }
01136 
01137     return alpha_b_Derivative;
01138 }

01310     KernelFunction* kernel = svm->getKernel();

01315     int a, ac = xc * cc;
01316     int k, kc = kernel->getParameterDimension();
01317     int pc = getParameterDimension();
01318     Array<double> der;
01319     double C = get_C();
01320     SIZE_CHECK(kc + 1 == pc);
01321 
01322     alphaDerivative.resize(ac, pc, false);
01323     alphaDerivative = 0.0;
01324 
01325     std::vector<unsigned int> fI;           // indices of free SVs
01326     std::vector<unsigned int> bI;           // indices of bounded SVs
01327     Array<bool> sv(xc);                     // is the EXAMPLE a SV?
01328     sv = false;
01329     a = 0;
01330     for (x=0; x<xc; x++)
01331     {
01332         for (c=0; c<cc; c++)
01333         {
01334             double alpha = svm->getParameter(a);
01335             if (alpha > 0.0)
01336             {
01337                 if (alpha < C)
01338                 {
01339                     fI.push_back(a);
01340                 }
01341                 else
01342                 {
01343                     bI.push_back(a);
01344                 }
01345                 sv(x) = true;
01346             }
01347             a++;
01348         }
01349     }
01350     unsigned int f, f2, fc = fI.size();
01351     unsigned int b, bc = bI.size();
01352 
01353     // compute required parts of K and dK
01354     Array<double> K(xc, xc);                    // kernel matrix
01355     Array<double> dK(xc, xc, kc);               // kernel derivatives
01356     for (x=0; x<xc; x++)
01357     {
01358         if (! sv(x)) continue;
01359         for (x2=0; x2<x; x2++)
01360         {
01361             if (! sv(x2)) continue;
01362             K(x, x2) = kernel->evalDerivative(points[x], points[x2], der);
01363             for (k=0; k<kc; k++) dK(x, x2, k) = der(k);
01364         }
01365         K(x, x) = kernel->evalDerivative(points[x], points[x], der);
01366         for (k=0; k<kc; k++) dK(x, x, k) = der(k);
01367     }
01368 
01369     // build matrices F and B
01370     Matrix F(fc, fc);
01371     Vector B1(fc);
01372     for (f=0; f<fc; f++)
01373     {
01374         unsigned int x = fI[f] / cc;
01375         unsigned int v = fI[f] % cc;
01376         unsigned int y = (unsigned int)target(x, 0);
01377         for (f2=0; f2<0; f2++)
01378         {
01379             unsigned int x2 = fI[f2] / cc;
01380             unsigned int v2 = fI[f2] % cc;
01381             unsigned int y2 = (unsigned int)target(x2, 0);
01382             double mod = 0.0;
01383             if (y == y2) mod += 0.5;
01384             if (y == v2) mod -= 0.5;
01385             if (y2 == v) mod -= 0.5;
01386             if (v == v2) mod += 0.5;
01387 
01388             F(f, f2) = F(f2, f) = mod * K(x, x2);
01389         }
01390         if (y != v) F(f, f) = K(x, x);
01391 
01392         double sum = 0.0;
01393         for (b=0; b<bc; b++)
01394         {
01395             unsigned int x2 = bI[b] / cc;
01396             unsigned int v2 = bI[b] % cc;
01397             unsigned int y2 = (unsigned int)target(x2, 0);
01398             double mod = 0.0;
01399             if (y == y2) mod += 0.5;
01400             if (y == v2) mod -= 0.5;
01401             if (y2 == v) mod -= 0.5;
01402             if (v == v2) mod += 0.5;
01403 
01404             sum += mod * K(x, x2);
01405         }
01406         B1(f) = sum;
01407     }
01408 
01409     // invert F
01410     Matrix invF = F.inverse();
01411 
01412     // compute derivative of alpha w.r.t. C
01413     Vector tmp = invF * B1;
01414     for (f=0; f<fc; f++) alphaDerivative(fI[f], 0) = -tmp(f);
01415     for (b=0; b<bc; b++) alphaDerivative(bI[b], 0) = 1.0;
01416 
01417     // compute derivative of alpha w.r.t. all kernel parameters
01418     for (k=0; k<kc; k++)
01419     {
