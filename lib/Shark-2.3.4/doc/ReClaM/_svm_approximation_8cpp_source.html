00643     for (i = 0; i < *mpNoExamplesOfApproximatedSVM; ++i)
00644     {
00645         Beta(i) = 0;
00646         for (k = 0; k < mNoSVs; ++k)
00647             Beta(i) += KzInvDotKzx(i, k) * AlphaY(k);
00648     }
00649 
00650     // the additional one is for the threshold
00651     mpApproximatedSVM->parameter.resize(*mpNoExamplesOfApproximatedSVM + 1, false);
00652 
00653 
00654     for (i = 0; i < *mpNoExamplesOfApproximatedSVM; ++i)
00655     {
00656         if (isnan(Beta(i)))
00657         {
00658             if (verbose) cout << " ... numerical problem; maybe kernel not appropriate for dataset !!!" << endl;
00659             return false;
00660         }
00661         else
00662             mpApproximatedSVM->setParameter(i, Beta(i));
00663     }
00664     return true;
00665 }

00667 // determine "optimal" offset for SVM approximation
00668 bool SvmApproximation::calcOffsetForReducedModel()
00669 {
00670     unsigned int i, j;
00671 
00672     SVM &originalSVM     = *mpSVM ;
00673     SVM &approximatedSVM = *mpApproximatedSVM;
00674 
00675     const Array<double>& xxOriginalSVM = originalSVM.getPoints();
00676     const Array<double>& xxApproximatedSVM = approximatedSVM.getPoints();
00677 
00678     double sum = 0;
00679     double b = mpSVM->getOffset();
00680 
00681     // calculate the mean over all differences between the original and the reduced SVM
00682     for (i = 0; i < mNoExamplesOfOrigSVM; ++i)
00683     {
00684         if (mpSVM->getAlpha(i) == 0)
00685             continue;
00686 
00687         for (j = 0; j < mNoExamplesOfOrigSVM; ++j)
00688             sum += mpSVM->getAlpha(j) * mpKernel->eval(xxOriginalSVM[j], xxOriginalSVM[i]);
00689 
00690         sum += b;
00691 
00692         for (j = 0; j < *mpNoExamplesOfApproximatedSVM; ++j)
00693             sum -= mpApproximatedSVM->getAlpha(j) * mpKernel->eval(xxApproximatedSVM[j],  xxOriginalSVM[i]);
00694     }
00695 
00696     mOffsetOfApproximatedSVM = sum / (double)this->mNoSVs;
00697 
00698     // set offset of approximated SVM
00699     mpApproximatedSVM->parameter(*mpNoExamplesOfApproximatedSVM) = mOffsetOfApproximatedSVM ;
00700 
00701     return true;
00702 }

00718

00752 void SvmApproximation::createIndexListWithStochasticUniversalSampling()
00753 {
00754     unsigned int i;
00755 
00756     SVM &originalSVM = *mpSVM ;
00757     double alphaSum = 0;
00758 
00759     for (i = 0; i < mNoExamplesOfOrigSVM; ++i)
00760         alphaSum += fabs(originalSVM.getAlpha(i));
00761 
00762     // create permutation vector
00763     std::vector<int> permVec(mNoExamplesOfOrigSVM);
00764     for (i = 0; i < mNoExamplesOfOrigSVM; ++i)
00765         permVec[i] = i;
00766 
00767     std::random_shuffle(permVec.begin(), permVec.end());
00768 
00769     float frac = (float)(alphaSum / (double)mTargetNoVecsForApproximatedSVM);
00770     Uniform abc(0.0, frac);
00771     double p = abc();
00772 
00773     if (mpIndexListOfVecsForSelection == NULL)
00774         mpIndexListOfVecsForSelection   = new(std::vector<int>);
00775 
00776     mpIndexListOfVecsForSelection->reserve(mTargetNoVecsForApproximatedSVM);
00777 
00778 
00779     double sum = 0;
00780     unsigned selectedElems = 0;
00781     i = 0;
00782 
00783     while (i < mNoExamplesOfOrigSVM)
00784     {
00785         sum +=  fabs(originalSVM.getAlpha(permVec[i]));
00786 
00787         while (sum >= p + selectedElems * frac)
00788         {
00789             selectedElems++;
00790             mpIndexListOfVecsForSelection->push_back(permVec[i]);
00791         }
00792         i++;
00793     }
00794 }

00883 // classify all original SVs with the approximated SVM
00884 float SvmApproximation::getClassificationRateOnSVsCorrectlyClassifiedByOrigSVM()
00885 {
00886     unsigned int i;
00887 
00888     Array<double> result(1, 1);
00889 
00890     unsigned correctlyClassified = 0;
00891     unsigned incorrectlyClassified = 0;
00892     unsigned noCorrectlyClassifiedByOriginalSVM = 0;
00893 
00894     for (i = 0; i < mNoExamplesOfOrigSVM; ++i)
00895     {
00896         if (mpSVM->getAlpha(i) != 0)
00897         {
00898             // prediction of original SVM
00899             mpSVM->model(mpSVM->getPoints().row(i), result);
00900 
00901             // SV not correctly classified by original SVM
00902             if (result(0)*labels[i] < 0)
00903                 continue;
00904 
00905             ++noCorrectlyClassifiedByOriginalSVM;
00906             mpApproximatedSVM->model(mpSVM->getPoints().row(i), result);
00907 
00908             if (result(0)*labels[i] > 0)
00909                 ++correctlyClassified;
00910             else
00911                 ++incorrectlyClassified;
00912         }
00913     }
00914 
00915     return (float)correctlyClassified / (float)noCorrectlyClassifiedByOriginalSVM;
00916 }

00920 {

00923     mTargetNoPositiveVectors    = (unsigned)((float)mNoPositiveSVsOfOrigSVM / (float)mNoSVs * (float)mTargetNoVecsForApproximatedSVM) ;

00926         mTargetNoPositiveVectors = 1;

00929         mTargetNoPositiveVectors = mTargetNoVecsForApproximatedSVM - 1;

00931     mTargetNoNegativeVectors =  mTargetNoVecsForApproximatedSVM - mTargetNoPositiveVectors;
