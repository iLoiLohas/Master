The []-operator

The []-operator returns a sub-Array, as shown in the following example. 

#include <Array/ArrayIo.h>

int main(int argc, char** argv)
{
    Array<double> m(3,2);

    for(unsigned i=0; i<m.dim(0); i++) 
      for(unsigned j=0; j<m.dim(1); j++) 
        m(i,j)=(i+j*m.dim(0))/10.;

    writeArray(m[1], std::cout);

    return EXIT_SUCCESS;
}

The program outputs the second row of the 3-times-2 matrix m:

0.1 0.4
