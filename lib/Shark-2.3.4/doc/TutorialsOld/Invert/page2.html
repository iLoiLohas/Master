Example Applications

The following code demonstrates how to invert a 3 times 3 matrix with the g_inverse method, resulting in the pseudoinverse. 

#include <LinAlg/LinAlg.h>
#include <Array/ArrayIo.h>

int main(int argc, char** argv)
{
    Array<double> matrix(3, 3);
    Array<double> inverse(3, 3);

    matrix(0, 0) = 1.0;
    matrix(0, 1) = -1.0;
    matrix(0, 2) = 0.0;
    matrix(1, 0) = 0.0;
    matrix(1, 1) = 3.0;
    matrix(1, 2) = -2.0;
    matrix(2, 0) = 0.0;
    matrix(2, 1) = 1.0;
    matrix(2, 2) = 1.0;

    g_inverse(matrix, inverse);

    std::cout << "matrix:" << std::endl;
    writeArray(matrix, std::cout);
    std::cout << "inverse:" << std::endl;
    writeArray(inverse, std::cout);
}

The program first prints the matrix and then the inverse:

matrix: 1 -1 0 0 3 -2 0 1 1 inverse: 1 0.2 0.4 9.18861e-18 0.2 0.4 1.37531e-16 -0.2 0.6

The two numbers very close to zero are numerical inaccuracies. These are in principle not avoidable. For a symmetric matrix, the function invertSymm works similar. The only difference is that the inverse is the first argument. 

#include <LinAlg/LinAlg.h>
#include <Array/ArrayIo.h>

int main(int argc, char** argv)
{
    Array<double> matrix(3, 3);
    Array<double> inverse(3, 3);

    matrix(0, 0) = 1.0;
    matrix(0, 1) = -1.0;
    matrix(0, 2) = 0.0;
    matrix(1, 0) = -1.0;
    matrix(1, 1) = 3.0;
    matrix(1, 2) = -2.0;
    matrix(2, 0) = 0.0;
    matrix(2, 1) = -2.0;
    matrix(2, 2) = 1.0;

    invertSymm(inverse, matrix);

    std::cout << "matrix:" << std::endl;
    writeArray(matrix, std::cout);
    std::cout << "inverse:" << std::endl;
    writeArray(inverse, std::cout);
}

The program outputs the result:

matrix: 1 -1 0 -1 3 -2 0 -2 1 inverse: 0.5 -0.5 -1 -0.5 -0.5 -1 -1 -1 -1
