02224 //===========================================================================
02225 //
02226 // Combination of EM and jack-knife estimation (replaces the original M step)
02227 //
02228 // according to Z. R. Yang & S. Chen (1998).
02229 //              Robust maximum likelihood training of heteroscedastic
02230 //              probabilistic neural networks.
02231 //              Neural Networks 11(4), pp. 739-747. 1998.
02232 //
02233 // For the sake of computational efficiency the variances are not estimated
02234 // directly but by the relation V[x] = E[x^2] - E[x]^2.
02235 // This allows the use of an one-pass estimation of the variances.
02236 // (This may be bought at the expense of a decreased robustness of the
02237 //  estimation.)
02238 //
02239 // (epsilon ignored)
02240 //
02241 void MixtureOfGaussians::jack_knife_em(const Array< double >& x,
02242                                        double   precision,
02243                                        unsigned maxIteration,
02244                                        double   minVariance,
02245                                        double   beta)
02246 {
02247     //SIZE_CHECK( x.ndim( ) == 2 )
02248 
02249     unsigned i, j, k, iteration;
02250     double   px, last, ll = -1e30;
02251 
02252     if (size() == 1) {
02253         Array< double > newM(dim());
02254         Array< double > newV(dim());
02255 
02256         for (i = dim(); i--;) {
02257             newM(i) = 0.;
02258             newV(i) = 0.;
02259         }
02260 
02261         for (k = x.dim(0); k--;)
02262             for (i = dim(); i--;) {
02263                 newM(i) += x(k, i);
02264                 newV(i) += Shark::sqr(x(k, i));
02265             }
02266 
02267         a(0) = 1.;
02268         for (i = dim(); i--;) {
02269             m(0, i) = newM(i) / x.dim(0);
02270             v(0, i) = newV(i) / x.dim(0) - Shark::sqr(m(0, i));
02271         }
02272 
02273         iteration = 0;
02274     }
02275     else {
02276         double diffa;
02277         Array< double > pi(size());
02278         Array< double > newA(size());
02279         Array< double > newM(size(), dim());
02280         Array< double > newV(size(), dim());
02281         Array< double > jackA(size());
02282         Array< double > jackM(size(), dim());
02283         Array< double > jackV(size(), dim());
02284 
02285         for (iteration = 0; iteration < maxIteration; ++iteration) {
02286             last = ll;
02287             ll   = 0.;
02288 
02289             newA = 0.;
02290             newM = 0.;
02291             newV = 0.;
02292 
02293             //
02294             // loop over all data vectors
02295             //
02296             for (k = x.dim(0); k--;) {
02297                 if (beta < 1)
02298                     for (px = 0., i = size(); i--;)
02299                         px += (pi(i) = pow(p(x[ k ], i) * a(i), beta));
02300                 else
02301                     for (px = 0., i = size(); i--;)
02302                         px += (pi(i) = p(x[ k ], i) * a(i));
02303 
02304                 ll += log(px = Shark::max(px, MIN_VAL));
02305 
02306                 for (i = size(); i--;) {
02307                     newA(i) += (pi(i) /= px);
02308                     for (j = dim(); j--;) {
02309                         newM(i, j) += pi(i) * x(k, j);
02310                         newV(i, j) += pi(i) * Shark::sqr(x(k, j));
02311                     }
02312                 }
02313             }
02314 
02315             //
02316             // loop over all data vectors again to build the
02317             // jack-knife estimate
02318             //
02319             for (i = size(); i--;) {
02320                 jackA(i) = newA(i);
02321                 for (j = dim(); j--;) {
02322                     jackM(i, j) = x.dim(0) * newM(i, j) / newA(i);
02323                     jackV(i, j) = x.dim(0) * newV(i, j) / newA(i);
02324                 }
02325             }
02326 
02327             for (k = x.dim(0); k--;) {
02328                 if (beta < 1)
02329                     for (px = 0., i = size(); i--;)
02330                         px += (pi(i) = pow(p(x[ k ], i) * a(i), beta));
02331                 else
02332                     for (px = 0., i = size(); i--;)
02333                         px += (pi(i) = p(x[ k ], i) * a(i));
02334 
02335                 px = Shark::max(px, MIN_VAL);
02336 
02337                 for (i = size(); i--;) {
02338                     diffa       = (newA(i) - (pi(i) /= px));
02339                     jackA(i) -= diffa / x.dim(0);
02340                     diffa      *= x.dim(0) / (x.dim(0) - 1.);
02341                     for (j = dim(); j--;) {
02342                         jackM(i, j) -= (newM(i, j) - pi(i) * x(k, j)) / diffa;
02343                         jackV(i, j) -= (newV(i, j) - pi(i) * Shark::sqr(x(k, j))) / diffa;
02344                     }
02345                 }
02346             }
02347 
02348             for (i = size(); i--;) {
02349                 a(i) = jackA(i);
02350                 for (j = dim(); j--;) {
02351                     m(i, j) = jackM(i, j);
02352                     v(i, j) = Shark::max(jackV(i, j) - Shark::sqr(m(i, j)),
02353                                        minVariance);
02354                 }
02355             }
02356 
02357             if (2 *(ll - last) / fabs(ll + last) < precision)
02358                 break;
02359         }
02360 
02361         normalize();
02362     }
02363 }
